# Commands to update and push changes to the homebrew-tap repository

cd /Users/jpb/workspace/homebrew-tap
git add Casks/chromeless.rb
git commit -m "Update Chromeless to v5.0.0 with functional DMG"
git push origin main

# After pushing the changes, users can update Chromeless via Homebrew:
brew update
brew upgrade --cask jonathanpberger/tap/chromeless

# To test a fresh install:
brew uninstall --cask jonathanpberger/tap/chromeless
brew install --cask jonathanpberger/tap/chromeless
